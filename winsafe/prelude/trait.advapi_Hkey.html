fn EnumKeyEx<'a>(
    self
) -> SysResult<Box<dyn Iterator<Item = SysResult<String>> + 'a>>

Returns an iterator over the names of the keys, which calls RegEnumKeyEx repeatedly.

Examples
use winsafe::prelude::*;
use winsafe::{co, HKEY};

let hkey = HKEY::CURRENT_USER.OpenKeyEx(
    "Control Panel",
    co::REG_OPTION::default(),
    co::KEY::READ,
)?;

for key_name in hkey.EnumKeyEx()? {
    let key_name = key_name?;
    println!("{}", key_name);
}

hkey.CloseKey()?;

fn EnumValue<'a>(
    self
) -> SysResult<Box<dyn Iterator<Item = SysResult<(String, REG)>> + 'a>>

Returns an iterator of the names and types of the values, which calls RegEnumValue repeatedly.

Examples
use winsafe::prelude::*;
use winsafe::{co, HKEY};

let hkey = HKEY::CURRENT_USER.OpenKeyEx(
    "Control Panel\\Appearance",
    co::REG_OPTION::default(),
    co::KEY::READ,
)?;

for value_and_type in hkey.EnumValue()? {
    let (value, reg_type) = value_and_type?;
    println!("{}, {}", value, reg_type);
}

hkey.CloseKey()?;

fn GetValue(self, sub_key: &str, value: &str) -> SysResult<RegistryValue>

RegGetValue method.

The data type will be automatically queried with a first call to RegGetValue.

Examples
use winsafe::prelude::*;
use winsafe::{HKEY, RegistryValue};

let val = HKEY::CURRENT_USER.GetValue(
    "Control Panel\\Mouse",
    "Beep",
)?;

match val {
    RegistryValue::Dword(n) => println!("Number u32: {}", n),
    RegistryValue::Qword(n) => println!("Number u64: {}", n),
    RegistryValue::Sz(str) => println!("String: {}", str),
    RegistryValue::Binary(bin) => {
        println!("Binary:");
        for b in bin.iter() {
            print!("{:02x} ", b);
        }
        println!("");
    },
    RegistryValue::None => println!("No value"),
}

fn SetValueEx(self, value: &str, data: RegistryValue) -> SysResult<()>

RegSetValueEx method.

If the value doesn’t exist, if will be created. If new type is different from current type, new type will prevail.

Examples

use winsafe::{co, HKEY, RegistryValue, WString};

let hkey = HKEY::CURRENT_USER.OpenKeyEx(
    "Console\\Git Bash",
    co::REG_OPTION::default(),
    co::KEY::ALL_ACCESS,

    "Color",
    RegistryValue::Sz(WString::from_str("blue")),
)?;
