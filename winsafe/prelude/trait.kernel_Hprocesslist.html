Prefer importing this trait through the prelude:

use winsafe::prelude::*;

fn iter<'a>(
    &'a self,
    pe32: &'a mut PROCESSENTRY32
) -> Box<dyn Iterator<Item = WinResult<&'a PROCESSENTRY32>> + 'a>

Returns an iterator over the processes list by calling HPROCESSLIST::Process32First and then HPROCESSLIST::Process32Next consecutively.

Examples
use winsafe::prelude::*;
use winsafe::{co, HPROCESSLIST, PROCESSENTRY32};

let mut pe = PROCESSENTRY32::default();
let hpl = HPROCESSLIST::
    CreateToolhelp32Snapshot(co::TH32CS::SNAPPROCESS, None)?;

for pe in hpl.iter(&mut pe) {
    let pe = pe?;
    println!("{} {} {}",
        pe.szExeFile(), pe.th32ProcessID, pe.cntThreads);
}

hpl.CloseHandle()?;
