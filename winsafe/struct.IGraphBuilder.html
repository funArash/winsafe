Expand description

IGraphBuilder COM interface over IGraphBuilderVT.

Automatically calls IUnknown::Release when the object goes out of scope.

Examples

use winsafe::prelude::*;
use winsafe::{co, CoCreateInstance, IGraphBuilder};

let obj = CoCreateInstance::<IGraphBuilder>(
    &co::CLSID::FilterGraph,
    None,
    co::CLSCTX::INPROC_SERVER,
)?;
