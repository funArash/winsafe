Examples

Selecting all text in the control:

use winsafe::prelude::*;

let my_edit: gui::Edit; // initialized somewhere

my_edit.set_selection(Some(0), None);

Clearing the selection:

use winsafe::gui;

let my_edit: gui::Edit; // initialized somewhere

my_edit.set_selection(None, None);
