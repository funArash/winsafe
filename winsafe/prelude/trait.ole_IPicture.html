fn get_Height(&self) -> HrResult<i32>

IPicture::get_Height method.

Note: Returns a value in HIMETRIC units. To convert it to pixels, prefer using the simpler IPicture::size_px, or use HDC::HiMetricToPixel to perform the conversion manually.

Examples
ⓘ
use winsafe::prelude::*;
use winsafe::{HWND, IPicture};

let pic: IPicture; // initialized somewhere

let hdc = HWND::NULL.GetDC()?;

let (_, height) = hdc.HiMetricToPixel(0, pic.get_Height()?);
println!("Height: {} px", height);

HWND::NULL.ReleaseDC(hdc)?;
