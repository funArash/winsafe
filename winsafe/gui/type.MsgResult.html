pub type MsgResult<T> = Result<T, MsgError>;
Available on crate feature gui only.
Expand description

A Result alias for errors from window message handling, which returns a MsgError on failure.

Examples

Converting into the generic AnyResult:

use winsafe::prelude::*;
use winsafe::{co, AnyResult, gui::MsgResult};

let run_result: MsgResult<()> = Ok(());

let err_result: AnyResult<()> = run_result.map_err(|err| err.into());
