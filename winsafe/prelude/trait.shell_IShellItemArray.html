use winsafe::prelude::*;

fn iter(&self) -> HrResult<Box<dyn Iterator<Item = HrResult<IShellItem>> + '_>>

Returns an iterator over the IShellItem elements by calling IShellItemArray::GetCount and IShellItemArray::GetItemAt consecutively.

Examples

Iterating over the IShellItem objects:

use winsafe::prelude::*;
use winsafe::{co, IShellItemArray};

let ish_arr: IShellItemArray; // initialized somewhere

for ish_item in ish_arr.iter()? {
    let ish_item = ish_item?;
    println!("Path: {}",
        ish_item.GetDisplayName(co::SIGDN::FILESYSPATH)?);
}

Collecting the file paths into a Vec:

use winsafe::prelude::*;
use winsafe::{co, HrResult, IShellItemArray};

let ish_arr: IShellItemArray; // initialized somewhere

let paths = ish_arr.iter()?
    .map(|shi|
        shi.and_then(|shi|
            shi.GetDisplayName(co::SIGDN::FILESYSPATH),
        ),
    )
    .collect::<HrResult<Vec<_>>>()?;
